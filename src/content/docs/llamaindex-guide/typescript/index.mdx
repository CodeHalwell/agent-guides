## Install, minimal example

```typescript
import { Workflow, StartEvent, StopEvent, step } from '@llamaindex/workflow';

class MyAgent extends Workflow {
  @step()
  async respond(ev: StartEvent): Promise<StopEvent> {
    return new StopEvent({ result: `Echo: ${ev.data.input}` });
  }
}

const workflow = new MyAgent();
const result = await workflow.run({ input: 'Hello!' });
console.log(result);
```
